18bool containsNonZero(const double* data, int size) {
19 // Allocate host memory
20 double* h_data = new double[size];
21
22 // Copy data from device to host
23 cudaMemcpy(h_data, data, size * sizeof(double), cudaMemcpyDeviceToHost);
24
25 bool hasNonZero = false;
26 // Check each element
27 for (int i = 0; i < size; ++i) {
28 if (h_data[i] != 0.0) {
29 hasNonZero = true;
30 break;
31 }
32 }
33
34 // Free host memory
35 delete[] h_data;
36 return hasNonZero;
37}
