23 bool isApproximatelyEqual(double a, double b, double epsilon = 1e-6) {
24 if (std::isinf(a) && std::isinf(b)) {
25 return (a > 0) == (b > 0);
26 }
27 if (std::abs(a) > DBL_MAX / 2 || std::abs(b) > DBL_MAX / 2) {
28 // For very large numbers, use a relative error
29 return std::abs(a - b) / std::max(std::abs(a), std::abs(b)) < epsilon;
30 }
31 return std::abs(a - b) < epsilon;
32 }
